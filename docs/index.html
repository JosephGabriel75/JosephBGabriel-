  geom_point() + 
  labs(x = "Sepal Width", y =  "Sepal Length") + 
  ggtitle("Setosa, Versicolor, and Virginica Measurements") + 
  theme(plot.title = element_text(hjust = .5)) + 
  geom_smooth() +
  scale_size_area()
